Code Example

Given a string, remove any characters that are unique from the string (JS): 

            
function onlyDuplicates(str) {
  const {duplicates} = [...str].reduce((acum, cur) => {
    if (acum.other.includes(cur)) {
        acum.duplicates.push(cur);
        return acum;
    }
    acum.other.push(cur);
    return acum;
  }, {"duplicates": [], "other": []});
  return [...str].filter((letter) => duplicates.includes(letter)).join('');
}
